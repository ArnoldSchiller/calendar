25 * ------------------------------------------------------------------
26 * Holiday data is stored in JSON files in the holidays/ directory:
27 *
28 * Structure:
29 * holidays/

45 * {
46 * "k": "f", // Type: 'f'=fixed, 'e'=easter-based, 'r'=relative

49 * "o": -2, // Offset from Easter (for easter-based)
50 * "n": "New Year", // Holiday name
51 * "p": true, // Public holiday (true) or observance (false)
52 * "c": "year<=1994" // Optional condition for historical changes
53 * }

57 * ------------------------------------------------------------------
58 * The system supports hierarchical regional holiday resolution:
59 *
60 * Example: User in Munich (Bavaria, Germany)

121/* ================================================================

133 /* ============================================================
134 * CONSTRUCTOR
135 * ============================================================

144 * Structure: { "de": [Holiday...], "de-BY": [Holiday...], ... }

147 * - JSON files are small (< 10KB each)

155 /* ============================================================
156 * HOLIDAY DATA LOADING
157 * ============================================================
158 *
159 * Loads holiday definitions based on system locale.
160 *
161 * LOCALE RESOLUTION LOGIC:
162 * 1. Get system locale (e.g., "de_DE.UTF-8")
163 * 2. Extract language code (e.g., "de")
164 * 3. Look for holidays/{lang}.json
165 * 4. Fallback: No holidays if file not found
166 *
167 * WHY AUTO-DETECT LOCALE?
168 * - Users don't need to configure their region
169 * - Works out-of-the-box for common locales
170 * - Can be extended with manual region selection

172 loadHolidays() {
173 try {
174 // Get system locale - GLib provides reliable locale detection
175 let locale = GLib.get_language_names()[0] || "en";
176 // Extract base language code (e.g., "de_DE" → "de")
177 let lang = locale.split('_')[0].split('.')[0].toLowerCase();
178 // Construct path to holiday definitions file

180 // Check if holiday file exists for this language

198 /* ============================================================
199 * PUBLIC API: GET HOLIDAYS FOR DATE
200 * ============================================================
201 *
202 * Primary public method used by UI components.
203 * Returns all holidays for a specific date and region.
204 *
205 * REGIONAL RESOLUTION LOGIC:
206 * 1. Get region-specific rules (e.g., "de-BY" for Bavaria)
207 * 2. Get base language rules (e.g., "de" for Germany)
208 * 3. Combine both lists, removing duplicates
209 *
210 * @param date - JavaScript Date object to check
211 * @param region - Regional code (e.g., "de-BY"). Defaults to "de"
212 * @returns Array of unique holiday names (empty if no holidays)
213 *
214 * USAGE EXAMPLES:
215 * - getHolidaysForDate(new Date(2026-01-01), "de") → ["New Year"]
216 * - getHolidaysForDate(new Date(2026-10-03), "de") → ["German Unity Day"]
217 * - getHolidaysForDate(new Date(2026-12-25), "de") → ["Christmas Day"]
218 */
219 getHolidaysForDate(date, region = "de") {
220 let dayHolidays = [];
221 // Get regional rules (e.g., Bavaria-specific)
222 let rules = this.holidayData[region] || [];
223 // Get base language rules (e.g., German national)
224 let baseRegion = region.split('-')[0];
225 let baseRules = this.holidayData[baseRegion] || [];
226 // Combine regional + base rules (regional takes precedence)
227 let allRules = rules.concat(baseRules);
228 // Check each rule against the target date
229 for (let rule of allRules) {
230 if (this.isHolidayMatch(rule, date)) {
231 dayHolidays.push(rule.n);
232 }
233 }
234 /**
235 * REMOVE DUPLICATES:
236 * Using a Set ensures that if a holiday (like "New Year") is defined in
237 * both the base and regional rules, it only appears once in the UI.
238 *
239 * WHY USE SET INSTEAD OF ARRAY CHECKS?
240 * - Simpler code
241 * - Better performance for small arrays
242 * - Maintains insertion order
243 */
244 return [...new Set(dayHolidays)];
245 }
246 /* ============================================================
247 * HOLIDAY RULE MATCHING ENGINE
248 * ============================================================
249 *
250 * Core matching logic for different holiday types.
251 *
252 * SUPPORTED HOLIDAY TYPES:
253 * 1. Fixed dates (k='f'): Christmas, New Year, etc.
254 * 2. Easter-based (k='e'): Good Friday, Pentecost, etc.
255 * 3. Relative dates (k='r'): Not implemented in current version
256 *
257 * @param rule - Holiday rule definition from JSON
258 * @param date - Date to check against
259 * @returns true if date matches the holiday rule
260 */
261 isHolidayMatch(rule, date) {
262 const year = date.getFullYear();
263 const month = date.getMonth() + 1; // Convert JS 0-index to 1-index
264 const day = date.getDate();
265 // Check year-based conditions first (e.g., historical changes)
266 if (rule.c && !this.checkCondition(rule.c, year)) {
267 return false;
268 }
269 // Fixed date holiday (e.g., Christmas on Dec 25)
270 if (rule.k === 'f') {
271 return rule.m === month && rule.d === day;
272 }
273 // Easter-based holiday (e.g., Good Friday = Easter - 2)
274 if (rule.k === 'e') {
275 let easter = this.getEaster(year);
276 let target = new Date(easter);
277 target.setDate(easter.getDate() + (rule.o || 0));
278 return target.getMonth() + 1 === month && target.getDate() === day;
279 }
280 // Relative date holiday (not yet implemented)
281 if (rule.k === 'r') {
282 // Future enhancement: "first Monday in September" etc.
283 return false;
284 }
285 return false;
286 }
287 /* ============================================================
288 * CONDITION PARSER
289 * ============================================================
290 *
291 * Parses simple year-based conditions for historical accuracy.
292 *
293 * SUPPORTED FORMATS:
294 * - "year<=1994" (holiday existed until 1994)
295 * - "year>=2000" (holiday started in 2000)
296 * - "year==1990" (holiday only in 1990)
297 *
298 * WHY NEED CONDITIONS?
299 * - Holidays change over time (political/religious)
300 * - German reunification (1990) changed many holidays
301 * - Some holidays were added/removed by law
302 *
303 * @param cond - Condition string from JSON
304 * @param year - Year to check
305 * @returns true if condition is satisfied
306 */
307 checkCondition(cond, year) {
308 const match = cond.match(/year([<>=!]+)(\d+)/);
309 if (!match || !match[1] || !match[2])
310 return true;
311 const operator = match[1];
312 const val = parseInt(match[2]);
313 switch (operator) {
314 case "<=": return year <= val;
315 case ">=": return year >= val;
316 case "==": return year === val;
317 case "<": return year < val;
318 case ">": return year > val;
319 default: return true;
320 }
321 }
322 /* ============================================================
323 * EASTER CALCULATION ALGORITHM
324 * ============================================================
325 *
326 * Calculates Easter Sunday date using the Meeus/Jones/Butcher algorithm.
327 * This is a refinement of Gauss's original algorithm.
328 *
329 * WHY IS EASTER COMPLICATED?
330 * - Based on lunar calendar (first Sunday after first full moon after vernal equinox)
331 * - Different churches use different calculations
332 * - This algorithm works for Gregorian calendar years 1583-4099
333 *
334 * ALGORITHM SOURCE:
335 * Jean Meeus, "Astronomical Algorithms", 2nd edition, 1998
336 *
337 * @param year - Gregorian calendar year
338 * @returns Date object for Easter Sunday
339 */
340 getEaster(year) {
341 // Step 1: Golden number - position in 19-year Metonic cycle
342 let a = year % 19;
343 // Step 2: Century number
344 let b = Math.floor(year / 100);
345 // Step 3: Years within century
346 let c = year % 100;
347 // Step 4: Leap year corrections
348 let d = Math.floor(b / 4);
349 let e = b % 4;
350 // Step 5: Correction for 30-year cycle
351 let f = Math.floor((b + 8) / 25);
352 // Step 6: Moon orbit correction
353 let g = Math.floor((b - f + 1) / 3);
354 // Step 7: Epact (age of moon on Jan 1)
355 let h = (19 * a + b - d - g + 15) % 30;
356 // Step 8: Weekday corrections
357 let i = Math.floor(c / 4);
358 let k = c % 4;
359 // Step 9: Correction for 7-day week
360 let l = (32 + 2 * e + 2 * i - h - k) % 7;
361 // Step 10: Final calculation for full moon
362 let m = Math.floor((a + 11 * h + 22 * l) / 451);
363 // Step 11: Month and day calculation
364 let n = h + l - 7 * m + 114;
365 let month = Math.floor(n / 31);
366 let day = (n % 31) + 1;
367 // Return Easter Sunday date (month is 1-indexed in algorithm)
368 return new Date(year, month - 1, day);
369 }
